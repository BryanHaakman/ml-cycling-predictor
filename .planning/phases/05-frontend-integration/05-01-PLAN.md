**Function 1: `loadFromPinnacle()`**

```javascript
async function loadFromPinnacle() {
    const btn = document.getElementById('load-pinnacle-btn');
    const err = document.getElementById('batch-error');
    const pickerWrap = document.getElementById('pinnacle-race-picker');
    const sel = document.getElementById('pinnacle-race-select');
    err.style.display = 'none';

    // Reset state per D-10
    _pinnacleRaces = null;
    _pinnacleMatchupIds = null;
    document.getElementById('refresh-odds-btn').disabled = true;

    // Loading state per D-07
    btn.disabled = true;
    btn.textContent = '\u23F3 Loading from Pinnacle...';
    pickerWrap.style.display = 'none';

    try {
        const res = await fetch('/api/pinnacle/load', {
            method: 'POST',
            headers: {'Content-Type': 'application/json'},
            body: JSON.stringify({})
        });
        const data = await res.json();

        // Error handling per D-17, D-18
        if (!res.ok) {
            if (data.type === 'auth_error') {
                const envVar = data.env_var || 'PINNACLE_SESSION_COOKIE';
                err.innerHTML = 'Session expired. To fix:<br>' +
                    '1. Open DevTools on pinnacle.ca \u2192 Network tab \u2192 any request<br>' +
                    '2. Copy the X-Session header value<br>' +
                    '3. Set ' + envVar + '=&lt;value&gt; in your environment<br>' +
                    '4. Restart Flask (python webapp/app.py)';
            } else {
                err.textContent = 'Load failed: ' + (data.error || 'Unknown error');
            }
            err.style.display = 'block';
            return;
        }

        if (!data.races || data.races.length === 0) {
            err.textContent = 'No cycling markets found for today. Check that your Pinnacle session is active and markets are open.';
            err.style.display = 'block';
            return;
        }

        // Store races
        _pinnacleRaces = data.races;

        // Populate race picker per D-04
        sel.innerHTML = '<option value="">\u2014 Select a Race \u2014</option>';
        data.races.forEach((race, i) => {
            const opt = document.createElement('option');
            opt.value = i;
            opt.textContent = race.race_name + ' (' + race.pairs.length + ' matchups)';
            sel.appendChild(opt);
        });
        pickerWrap.style.display = '';

        // Per D-04: if single race, auto-select immediately
        if (data.races.length === 1) {
            sel.value = '0';
            populatePinnacleRace(0);
        }
    } catch (e) {
        err.textContent = 'Load failed: ' + e.message;
        err.style.display = 'block';
    } finally {
        btn.disabled = false;
        btn.textContent = 'Load from Pinnacle';
    }
}
```

**Function 2: `onPinnacleRaceChange()`** (thin wrapper called by select onchange)

```javascript
function onPinnacleRaceChange() {
    const sel = document.getElementById('pinnacle-race-select');
    const idx = parseInt(sel.value);
    if (isNaN(idx) || !_pinnacleRaces || !_pinnacleRaces[idx]) return;
    populatePinnacleRace(idx);
}
```

**Function 3: `populatePinnacleRace(idx)`**

```javascript
function populatePinnacleRace(idx) {
    const race = _pinnacleRaces[idx];
    if (!race) return;

    // Per D-09: always replace stage fields
    const sc = race.stage_context || {};
    document.getElementById('batch-race-name').value = race.race_name || '';
    document.getElementById('batch-race-date').value = sc.race_date || '';
    document.getElementById('batch-distance').value = sc.distance || '';
    document.getElementById('batch-vert').value = sc.vertical_meters || '';
    document.getElementById('batch-profile').value = sc.profile_icon || 'p3';
    document.getElementById('batch-stage-type').value = sc.stage_type || 'RR';
    document.getElementById('batch-oneday').value = sc.is_one_day_race ? '1' : '0';
    document.getElementById('batch-climbs').value = sc.num_climbs || '';
    document.getElementById('batch-uci-tour').value = sc.uci_tour || '';
    document.getElementById('batch-race-url').value = sc.race_base_url || '';

    // Per D-08: clear existing pairs, replace with Pinnacle pairs
    clearBatchPairs();
    _pinnacleMatchupIds = [];

    race.pairs.forEach(pair => {
        addBatchPair();
        const idx = batchPairCounter - 1;

        // Store matchup_id per D-10
        _pinnacleMatchupIds.push({matchup_id: pair.matchup_id, row_idx: idx});

        // Rider A
        const aInput = document.getElementById('bp-a-input-' + idx);
        const aUrl = document.getElementById('bp-a-url-' + idx);
        if (pair.rider_a_resolved) {
            aInput.value = pair.pinnacle_name_a;
            aUrl.value = pair.rider_a_url;
        } else {
            // Per D-14: pre-fill with best_candidate hint if available
            if (pair.best_candidate_a_name) {
                aInput.value = pair.best_candidate_a_name;
                aUrl.value = pair.best_candidate_a_url || '';
            } else {
                aInput.value = '';
                aUrl.value = '';
            }
            // Per D-15: orange border for unresolved
            const aWrap = aInput.closest('.autocomplete-wrap');
            if (aWrap) aWrap.parentElement.classList.add('unresolved-rider');

            // Clear orange when user selects a valid rider (URL populated)
            const aObserver = new MutationObserver(() => {
                if (aUrl.value) {
                    aWrap.parentElement.classList.remove('unresolved-rider');
                    aObserver.disconnect();
                }
            });
            aObserver.observe(aUrl, {attributes: true, attributeFilter: ['value']});
            // Also listen for programmatic .value changes via input event on the hidden field
            aUrl.addEventListener('change', function handler() {
                if (aUrl.value) {
                    aWrap.parentElement.classList.remove('unresolved-rider');
                    aUrl.removeEventListener('change', handler);
                }
            });
        }

        // Rider B
        const bInput = document.getElementById('bp-b-input-' + idx);
        const bUrl = document.getElementById('bp-b-url-' + idx);
        if (pair.rider_b_resolved) {
            bInput.value = pair.pinnacle_name_b;
            bUrl.value = pair.rider_b_url;
        } else {
            if (pair.best_candidate_b_name) {
                bInput.value = pair.best_candidate_b_name;
                bUrl.value = pair.best_candidate_b_url || '';
            } else {
                bInput.value = '';
                bUrl.value = '';
            }
            const bWrap = bInput.closest('.autocomplete-wrap');
            if (bWrap) bWrap.parentElement.classList.add('unresolved-rider');

            const bObserver = new MutationObserver(() => {
                if (bUrl.value) {
                    bWrap.parentElement.classList.remove('unresolved-rider');
                    bObserver.disconnect();
                }
            });
            bObserver.observe(bUrl, {attributes: true, attributeFilter: ['value']});
            bUrl.addEventListener('change', function handler() {
                if (bUrl.value) {
                    bWrap.parentElement.classList.remove('unresolved-rider');
                    bUrl.removeEventListener('change', handler);
                }
            });
        }

        // Odds per D-11: set data-source='auto'
        const oddsA = document.getElementById('bp-odds-a-' + idx);
        const oddsB = document.getElementById('bp-odds-b-' + idx);
        oddsA.value = pair.odds_a;
        oddsB.value = pair.odds_b;
        oddsA.setAttribute('data-source', 'auto');
        oddsB.setAttribute('data-source', 'auto');

        // Per D-12: oninput switches data-source to 'user'
        oddsA.addEventListener('input', function() { this.setAttribute('data-source', 'user'); });
        oddsB.addEventListener('input', function() { this.setAttribute('data-source', 'user'); });
    });

    // Enable Refresh Odds per D-03
    document.getElementById('refresh-odds-btn').disabled = false;
    updateBatchPairCount();
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser JS -> Flask API | Frontend sends POST to /api/pinnacle/load and /api/pinnacle/refresh-odds; responses are untrusted JSON |
| Flask API -> Browser DOM | API response data injected into DOM via .value and .textContent (not innerHTML for user data) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-01 | Spoofing | /api/pinnacle/load | mitigate | Endpoint already protected by `_require_localhost` decorator (Phase 4); no additional auth needed in frontend |
| T-05-02 | Tampering | DOM injection via API response | mitigate | Use `.value` and `.textContent` for all data population (never `.innerHTML` with API data); race names in select options use `.textContent` property |
| T-05-03 | Information Disclosure | Session cookie in browser | accept | Cookie is only in server-side env var, never sent to or visible in browser JS; frontend only triggers the load, server handles auth |
| T-05-04 | Denial of Service | Large /load response | accept | Response size bounded by Pinnacle's active markets (~65 matchups max confirmed); no amplification possible |
| T-05-05 | Tampering | Auth error message | mitigate | Auth error message uses `.innerHTML` with hardcoded template strings only; `env_var` field from response is inserted via string concatenation but contains only a known env var name (`PINNACLE_SESSION_COOKIE`); the value is never user-controlled input from external sources |
</threat_model>
