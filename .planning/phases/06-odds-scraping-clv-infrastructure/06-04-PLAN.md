    ```javascript
    // When Book Bet clicked:
    const row = /* get the pair data from the row index */;
    const stakeInput = row.querySelector('.stake-input');
    const stake = parseFloat(stakeInput.value);
    if (isNaN(stake) || stake <= 0) {
      alert('Enter a valid stake amount');
      return;
    }
    const selectionName = /* determine which rider is the value bet based on edge sign */;
    const msg = `Book ${selectionName} @ ${odds.toFixed(2)} — Stake $${stake.toFixed(2)} CAD (1/4 Kelly). This cannot be undone.`;
    if (!window.confirm(msg)) return;

    // Disable button, show "Booking..."
    btn.disabled = true;
    btn.textContent = 'Booking...';

    fetch('/api/pnl/bet', {
      method: 'POST',
      headers: {'Content-Type': 'application/json'},
      body: JSON.stringify({
        stage_url: raceData.stage_context.stage_url || '',
        race_name: raceData.race_name,
        race_date: raceData.stage_context.race_date || '',
        rider_a_url: pair.rider_a_url,
        rider_a_name: pair.pinnacle_name_a,
        rider_b_url: pair.rider_b_url,
        rider_b_name: pair.pinnacle_name_b,
        selection: /* 'A' or 'B' based on which rider has positive edge */,
        decimal_odds: /* the odds for the selected rider */,
        model_prob: pair.model_prob,
        kelly_fraction: pair.edge / (pair.odds - 1),  // simplified Kelly
        stake: stake,
        recommended_stake: pair.recommended_stake,
        model_used: 'CalibratedXGBoost',
        is_one_day_race: raceData.stage_context.is_one_day_race ? 1 : 0,
        stage_type: raceData.stage_context.stage_type || 'RR',
        profile_icon: raceData.stage_context.profile_icon || '',
        distance_km: raceData.stage_context.distance_km || 0,
        vertical_meters: raceData.stage_context.vertical_meters || 0,
        num_climbs: raceData.stage_context.num_climbs || 0,
      })
    })
    .then(r => r.json())
    .then(data => {
      if (data.error) {
        // Show inline error below button in --red color
        showInlineError(btn, data.error);
        btn.disabled = false;
        btn.textContent = 'Book Bet';
      } else {
        btn.textContent = 'Booked!';
        btn.classList.add('btn-success');
        setTimeout(() => {
          btn.disabled = true;
          btn.textContent = 'Booked';
          btn.style.opacity = '0.4';
        }, 2000);
      }
    })
    .catch(err => {
      showInlineError(btn, 'Bet could not be saved. Check your connection and try again. No bet was recorded.');
      btn.disabled = false;
      btn.textContent = 'Book Bet';
    });
    ```
