          document.getElementById('clv-avg').textContent = (data.avg_clv * 100).toFixed(1) + '%';
          document.getElementById('clv-avg').style.color = data.avg_clv >= 0 ? 'var(--green)' : 'var(--red)';
          document.getElementById('clv-no-vig').textContent = (data.avg_clv_no_vig * 100).toFixed(1) + '%';

          document.getElementById('clv-n-bets').textContent = data.n_bets;
        }
      });

    // Fetch terrain CLV
    fetch('/api/pnl/clv-by-terrain')
