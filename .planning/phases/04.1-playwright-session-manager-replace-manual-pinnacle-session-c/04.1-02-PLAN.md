```python
"""
Playwright-based Pinnacle session acquisition with adaptive TTL.

Automates login to pinnacle.ca via headless Chromium, extracts the X-Session
token from response headers or cookies, and caches it with an adaptive TTL
that learns from real session expiry events.

Auth acquisition order (called from data/odds.py._get_api_key):
  1. PINNACLE_SESSION_COOKIE env var override (checked in odds.py, not here)
  2. Disk cache + adaptive TTL check -> return cached token if fresh
  3. Playwright headless login -> extract token -> update cache + TTL
  4. JS bundle extraction fallback (handled in odds.py, not here)

Public interface:
  - get_session_token()                  -> str or None (cache -> Playwright -> None)
  - invalidate_session()                 -> None (update TTL, clear cached token)
  - acquire_session_token(headless=True) -> str (Playwright login)
  - acquire_session_token_with_fallback()-> str (headless first, headed on CAPTCHA)
  - CaptchaOrMfaDetected                 -> Exception subclass
"""

import json
import logging
import os
import time
from typing import Optional

from config import SESSION_STATE_PATH, PINNACLE_HOME_URL

log = logging.getLogger(__name__)

# Sentinel for "no TTL learned yet" — ~115 days, far longer than any real session
_TTL_INFINITY_SENTINEL: int = 9_999_999
```

Key implementation details:

**`_load_session_state() -> dict`**: Read `SESSION_STATE_PATH`. Return `{"token": None, "last_acquired": 0.0, "learned_ttl_seconds": 9999999}` as defaults. Catch `OSError` and `json.JSONDecodeError`, return defaults on failure.

**`_save_session_state(state: dict) -> None`**: Write to `SESSION_STATE_PATH`. Ensure `learned_ttl_seconds` is stored as integer (never `float("inf")`). Catch `OSError` and log warning.

**`_is_session_fresh(state: dict) -> bool`**: Return `True` if `state["token"]` is truthy AND either TTL is the infinity sentinel OR `time.time() - state["last_acquired"] < state["learned_ttl_seconds"]`.

**`class CaptchaOrMfaDetected(Exception)`**: Simple exception for CAPTCHA/2FA detection.

**`acquire_session_token(headless: bool = True) -> str`**: Per D-01:
- Use `from playwright.sync_api import sync_playwright, TimeoutError as PlaywrightTimeoutError` (import inside function to avoid import-time dependency when mocking)
- `with sync_playwright() as p:` context manager
- Launch `p.chromium.launch(headless=headless)`
- Create context, new page
- Register `page.on("response", on_response)` handler to capture `x-session` header (lowercase!) from response headers. Important: keep the LAST token seen (overwrite), not the first — per RESEARCH.md Pitfall 4.
- Navigate to `PINNACLE_HOME_URL` (`https://www.pinnacle.ca/`)
- Fill username: `page.get_by_label("Username").fill(os.environ["PINNACLE_USERNAME"])` — read from env at call time, NOT module level (per Pitfall 6)
- Fill password: `page.get_by_label("Password").fill(os.environ["PINNACLE_PASSWORD"])`
- Click sign in: `page.get_by_role("button", name="Sign in").click()`
- Wait for "I'm fit to play" responsible gambling button and click it. Use text-based selector: `page.get_by_text("I'm fit to play").click()` with timeout 15000. If the text differs, try `page.get_by_role("button", name="fit to play")`.
- Wait for post-login indicator with timeout 15000ms. Try `page.wait_for_url("**/en/sports/**", timeout=15000)` or a balance element. On `PlaywrightTimeoutError`, raise `CaptchaOrMfaDetected("Post-login element not found")`.
- Cookie fallback: if `captured_token` dict has no value, iterate `context.cookies()` looking for cookie name containing "session" (case-insensitive).
- `browser.close()`
- If no token found, raise `PinnacleAuthError` (import from `data.odds`)
- Return token string

**`acquire_session_token_with_fallback() -> str`**: Per D-01:
- Try `acquire_session_token(headless=True)`
- Catch `CaptchaOrMfaDetected`, log warning, retry with `acquire_session_token(headless=False)`

**`get_session_token() -> Optional[str]`**: Per D-09, D-10:
- Load state from disk via `_load_session_state()`
- If `_is_session_fresh(state)`, return `state["token"]`
- Otherwise, try `acquire_session_token_with_fallback()`
- On success: update state with new token, `time.time()` as `last_acquired`, preserve existing `learned_ttl_seconds`
- `_save_session_state(state)`, return token
- On failure (any exception): log, return `None`

**`invalidate_session() -> None`**: Per D-12:
- Load state
- Compute `age = time.time() - state["last_acquired"]`
- If `age > 0` and `age < state["learned_ttl_seconds"]`: update `learned_ttl_seconds` to `age` (TTL only shrinks per D-07)
- Set `state["token"] = None`
- Save state
- Log the new TTL value

Use 2-space indentation. Type hints on all functions. Docstrings on all public functions. `log = logging.getLogger(__name__)`.

```python
def _get_api_key() -> str:
  """Resolve the Pinnacle auth token using the lookup chain.

  Lookup order (per D-10):
    1. PINNACLE_SESSION_COOKIE environment variable — manual override (D-05)
    2. Session cache + adaptive TTL check via get_session_token() (D-07, D-09)
    3. Playwright session acquisition via get_session_token() (D-01)
    4. JS bundle extraction via _extract_key_from_bundle() (last resort)
    5. All paths exhausted -> raise PinnacleAuthError

  Returns:
    The API key or session token string (stripped of whitespace).

  Raises:
    PinnacleAuthError: If all lookup paths are exhausted.
  """
  # 1. Environment variable override (D-05 — highest priority escape hatch)
  env_key = os.environ.get("PINNACLE_SESSION_COOKIE", "").strip()
  if env_key:
    return env_key

  # 2+3. Session manager: cached token or Playwright acquisition (D-07, D-09, D-01)
  from data.session_manager import get_session_token
  session_token = get_session_token()
  if session_token:
    return session_token

  # 4. Disk cache (existing .pinnacle_key_cache from JS bundle extraction)
  if os.path.exists(KEY_CACHE_PATH):
    try:
      with open(KEY_CACHE_PATH, "r", encoding="utf-8") as f:
        cached = f.read().strip()
      if cached:
        log.info("_get_api_key: using cached key from %s", KEY_CACHE_PATH)
        return cached
    except OSError as e:
      log.warning("_get_api_key: could not read key cache: %s", e)

  # 5. JS bundle extraction (last resort)
  log.info("_get_api_key: no session token — attempting JS bundle extraction")
  extracted = _extract_key_from_bundle()
  if extracted:
    try:
      with open(KEY_CACHE_PATH, "w", encoding="utf-8") as f:
        f.write(extracted)
      log.info("_get_api_key: wrote extracted key to cache %s", KEY_CACHE_PATH)
    except OSError as e:
      log.warning("_get_api_key: could not write key cache: %s", e)
    return extracted

  # 6. All paths exhausted
  raise PinnacleAuthError(
    "Pinnacle API key could not be obtained. "
    "Set the PINNACLE_SESSION_COOKIE environment variable as a manual override, "
    "or add PINNACLE_USERNAME and PINNACLE_PASSWORD to .env for automated login."
  )
```

Note: Use a local import `from data.session_manager import get_session_token` inside the function body to avoid circular import issues and to avoid importing Playwright at module load time.

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| .env -> os.environ | Pinnacle credentials loaded from disk |
| Playwright -> pinnacle.ca | Credentials sent to Pinnacle login page over HTTPS |
| pinnacle.ca -> session_manager | Session token extracted from response headers/cookies |
| session_state.json -> disk | Token stored in plaintext on local filesystem |

| Threat ID | Category | Component | Disposition | Mitigation Plan |
