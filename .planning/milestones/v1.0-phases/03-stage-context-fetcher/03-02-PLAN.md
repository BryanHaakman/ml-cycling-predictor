```python
@pytest.mark.integration
class TestLiveIntegration:
    """Live integration tests -- requires PCS network access."""

    def test_known_race_resolves(self):
        """Test that a known WT race in cache.db resolves to a StageContext."""
        # Use a race name that is guaranteed to be in cache.db
        # "Tour de Romandie" is in cache.db (verified during research)
        # Even if today's stage doesn't exist, the race should resolve
        # and _do_fetch should either find today's stage or return unresolved
        result = fetch_stage_context("Tour de Romandie - Stage 1")
        # The race name should fuzzy-match; whether we get is_resolved=True
        # depends on whether a stage matches today's date
        assert isinstance(result, StageContext)
        # If resolved, check field quality
        if result.is_resolved:
            assert result.distance > 0.0
            assert result.profile_icon in ("p1", "p2", "p3", "p4", "p5")
            assert result.stage_type in ("RR", "ITT", "TTT")
            assert result.race_date != ""
            assert result.uci_tour != ""
            assert result.race_base_url.startswith("race/")

    def test_unresolvable_race_degrades(self):
        """Test that a bogus race name returns is_resolved=False quickly."""
        import time
        start = time.time()
        result = fetch_stage_context("ZZZZZ Completely Fake Race 9999")
        elapsed = time.time() - start
        assert result.is_resolved is False
        assert elapsed < 6.0, f"Degradation took {elapsed:.1f}s, expected < 6s"

    def test_one_day_race_resolves(self):
        """Test a known one-day race (Paris-Roubaix) resolution."""
        result = fetch_stage_context("Paris-Roubaix")
        assert isinstance(result, StageContext)
        # Paris-Roubaix is in cache.db -- should at least fuzzy-match
        # is_resolved depends on whether today is race day
        if result.is_resolved:
            assert result.is_one_day_race is True
            assert result.distance > 0.0
```
